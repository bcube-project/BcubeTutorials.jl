function compute_residual(_u, V, params, cache)

    nŒì_perio_y = get_face_normals(dŒì_perio_y)

        -‚à´(flux_Œì(u, v, nŒì))dŒì +

    Œª4 = Œª1 - cAv

    duv2 = uAv * fdc1 + ny * fdc2 + (uAv / cAv + nx) * fdc3 + (uAv / cAv - nx) * fdc4
    duv3 = vAv * fdc1 - nx * fdc2 + (vAv / cAv + ny) * fdc3 + (vAv / cAv - ny) * fdc4
    duv4 =

        (uq41 - uq42) * fdc4

    return (
        Œ¥Œª_œÅ1 * (flu11 + duv1) +
        Œ¥Œª_œÅu1 ‚ãÖ (SA[flu21 + duv2, flu31 + duv3]) +

"""

"""

`f(q, t)` is the function to integrate.

function rk3_ssp(q, f::Function, t, Œît)
    stepper(q, t) = forward_euler(q, f, t, Œît)
    _q0 = get_dof_values(q)

    _q1 = stepper(q, Œît)

    set_dof_values!(q, _q1)
    _q2 = (3 / 4) .* _q0 .+ (1 / 4) .* stepper(q, t + Œît)

    set_dof_values!(q, _q2)
    _q1 .= (1 / 3) * _q0 .+ (2 / 3) .* stepper(q, t + Œît / 2)

    return _q1
end
