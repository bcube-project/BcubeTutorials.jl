function compute_residual(qdof, Q, V, params)
    q = (FEFunction(Q, qdof)...,)

    # alias on measures
    dŒì = params.dŒì
    dŒ© = params.dŒ©
    dŒì_wall = params.dŒì_wall
    dŒì_farfield = params.dŒì_farfield

    # Allocate rhs vectors
    b_vol = zero(qdof)
    b_fac = zero(qdof)

    # compute volume residuals
    l_vol(v) = ‚à´(flux_Œ©(q, v))dŒ©
    assemble_linear!(b_vol, l_vol, V)

    # face normals for each face domain (lazy, no computation at this step)
    nŒì = get_face_normals(dŒì)
    nŒì_wall = get_face_normals(dŒì_wall)
    nŒì_farfield = get_face_normals(dŒì_farfield)

    # flux residuals from interior faces for all variables
    l_Œì(v) = ‚à´(flux_Œì(q, v, nŒì))dŒì
    assemble_linear!(b_fac, l_Œì, V)

    # flux residuals from bc faces for all variables
    l_Œì_wall(v) = ‚à´(flux_Œì_wall(q, v, nŒì_wall))dŒì_wall
    l_Œì_farfield(v) = ‚à´(flux_Œì_farfield(q, v, nŒì_farfield))dŒì_farfield
    assemble_linear!(b_fac, l_Œì_wall, V)
    assemble_linear!(b_fac, l_Œì_farfield, V)
    dQ = b_vol .- b_fac

    return dQ
end

"""
    flux_Œ©(q, v)

Compute volume residual using the lazy-operators approach
"""
flux_Œ©(q, v) = _flux_Œ© ‚àò (q, map(‚àá, v))

function _flux_Œ©(q, ‚àáv)
    œÅ, œÅu, œÅE = q
    ‚àáŒª_œÅ, ‚àáŒª_œÅu, ‚àáŒª_œÅE = ‚àáv
    Œ≥ = stateInit.Œ≥

    vel = œÅu ./ œÅ
    œÅuu = œÅu * transpose(vel)
    p = pressure(œÅ, œÅu, œÅE, Œ≥)

    flux_œÅ  = œÅu
    flux_œÅu = œÅuu + p * I
    flux_œÅE = (œÅE + p) .* vel

    return return ‚àáŒª_œÅ ‚ãÖ flux_œÅ + ‚àáŒª_œÅu ‚ä° flux_œÅu + ‚àáŒª_œÅE ‚ãÖ flux_œÅE
end

"""
    flux_Œì(q, v, n)

Flux at the interface is defined by a composition of two functions:
* the input states at face sides which are needed for the riemann flux
* `flux_roe` defines the Riemann flux (as usual)
"""
flux_Œì(q, v, n) = flux_roe ‚àò (side‚Åª(q), side‚Å∫(q), jump(v), side‚Åª(n))

"""
    flux_roe(q‚Åª, q‚Å∫, Œ¥v, n)
"""
function flux_roe(q‚Åª, q‚Å∫, Œ¥v, n)
    Œ≥ = stateInit.Œ≥
    nx, ny = n
    œÅ1, (œÅu1, œÅv1), œÅE1 = q‚Åª
    œÅ2, (œÅu2, œÅv2), œÅE2 = q‚Å∫
    Œ¥Œª_œÅ1, Œ¥Œª_œÅu1, Œ¥Œª_œÅE1 = Œ¥v

    œÅ1 = max(eps(œÅ1), œÅ1)
    œÅ2 = max(eps(œÅ2), œÅ2)

    # Closure
    u1 = œÅu1 / œÅ1
    v1 = œÅv1 / œÅ1
    u2 = œÅu2 / œÅ2
    v2 = œÅv2 / œÅ2
    p1 = pressure(œÅ1, SA[œÅu1, œÅv1], œÅE1, Œ≥)
    p2 = pressure(œÅ2, SA[œÅu2, œÅv2], œÅE2, Œ≥)

    H2 = (Œ≥ / (Œ≥ - 1)) * p2 / œÅ2 + (u2 * u2 + v2 * v2) / 2.0
    H1 = (Œ≥ / (Œ≥ - 1)) * p1 / œÅ1 + (u1 * u1 + v1 * v1) / 2.0

    R = ‚àö(œÅ1 / œÅ2)
    invR1 = 1.0 / (R + 1)
    uAv = (R * u1 + u2) * invR1
    vAv = (R * v1 + v2) * invR1
    Hav = (R * H1 + H2) * invR1
    cAv = ‚àö(abs((Œ≥ - 1) * (Hav - (uAv * uAv + vAv * vAv) / 2.0)))
    ecAv = (uAv * uAv + vAv * vAv) / 2.0

    Œª1 = nx * uAv + ny * vAv
    Œª3 = Œª1 + cAv
    Œª4 = Œª1 - cAv

    d1 = œÅ1 - œÅ2
    d2 = œÅ1 * u1 - œÅ2 * u2
    d3 = œÅ1 * v1 - œÅ2 * v2
    d4 = œÅE1 - œÅE2

    # computation of the centered part of the flux
    flux_œÅ  = nx * œÅ2 * u2 + ny * œÅ2 * v2
    flux_œÅu = nx * p2 + flux_œÅ * u2
    flux_œÅv = ny * p2 + flux_œÅ * v2
    flux_œÅE = H2 * flux_œÅ

    # Temp variables
    rc1 = (Œ≥ - 1) / cAv
    rc2 = (Œ≥ - 1) / cAv / cAv
    uq41 = ecAv / cAv + cAv / (Œ≥ - 1)
    uq42 = nx * uAv + ny * vAv

    fdc1 = max(Œª1, 0.0) * (d1 + rc2 * (-ecAv * d1 + uAv * d2 + vAv * d3 - d4))
    fdc2 = max(Œª1, 0.0) * ((nx * vAv - ny * uAv) * d1 + ny * d2 - nx * d3)
    fdc3 =
        max(Œª3, 0.0) * (
            (-uq42 * d1 + nx * d2 + ny * d3) / 2.0 +
            rc1 * (ecAv * d1 - uAv * d2 - vAv * d3 + d4) / 2.0
        )
    fdc4 =
        max(Œª4, 0.0) * (
            (uq42 * d1 - nx * d2 - ny * d3) / 2.0 +
            rc1 * (ecAv * d1 - uAv * d2 - vAv * d3 + d4) / 2.0
        )

    duv1 = fdc1 + (fdc3 + fdc4) / cAv
    duv2 = uAv * fdc1 + ny * fdc2 + (uAv / cAv + nx) * fdc3 + (uAv / cAv - nx) * fdc4
    duv3 = vAv * fdc1 - nx * fdc2 + (vAv / cAv + ny) * fdc3 + (vAv / cAv - ny) * fdc4
    duv4 =
        ecAv * fdc1 +
        (ny * uAv - nx * vAv) * fdc2 +
        (uq41 + uq42) * fdc3 +
        (uq41 - uq42) * fdc4

    flux_œÅ  += duv1
    flux_œÅu += duv2
    flux_œÅv += duv3
    flux_œÅE += duv4

    return (Œ¥Œª_œÅ1 ‚ãÖ flux_œÅ + Œ¥Œª_œÅu1 ‚ãÖ SA[flux_œÅu, flux_œÅv] + Œ¥Œª_œÅE1 ‚ãÖ flux_œÅE)
end

function _flux_Œì_wall(q‚Åª, v‚Åª, n)
    Œ≥ = stateInit.Œ≥
    œÅ1, œÅu1, œÅE1 = q‚Åª
    Œª_œÅ1, Œª_œÅu1, Œª_œÅE1 = v‚Åª

    p1 = pressure(œÅ1, œÅu1, œÅE1, Œ≥)

    flux_œÅ  = zero(œÅ1)
    flux_œÅu = p1 * n
    flux_œÅE = zero(œÅE1)

    return (Œª_œÅ1 ‚ãÖ flux_œÅ + Œª_œÅu1 ‚ãÖ flux_œÅu + Œª_œÅE1 ‚ãÖ flux_œÅE)
end

function init!(q, dŒ©, initstate)
    AoA  = initstate.AoA
    Minf = initstate.M_inf
    Pinf = initstate.P_inf
    Tinf = initstate.T_inf
    r    = initstate.r_gas
    Œ≥    = initstate.Œ≥

    œÅinf = Pinf / r / Tinf
    ainf = ‚àö(Œ≥ * r * Tinf)
    Vinf = Minf * ainf
    œÅVxinf = œÅinf * Vinf * cos(AoA)
    œÅVyinf = œÅinf * Vinf * sin(AoA)
    œÅEinf = Pinf / (Œ≥ - 1) + 0.5 * œÅinf * Vinf^2

    œÅ0  = PhysicalFunction(x -> œÅinf)
    œÅu0 = PhysicalFunction(x -> SA[œÅVxinf, œÅVyinf])
    œÅE0 = PhysicalFunction(x -> œÅEinf)
    projection_l2!(q, (œÅ0, œÅu0, œÅE0), dŒ©)
    return nothing
end

function main(stateInit, stateBcFarfield, degree)
    @show degree, degquad

    meshpath = joinpath(
        dir,
        "..",
        "..",
        "input",
        "mesh",
        "naca0012_o" * string(mesh_degree) * ".msh",
    )
    mesh = read_mesh(meshpath)
    scale!(mesh, 1.0 / 0.5334)

    dimcar = compute_dimcar(mesh)

    DMPrelax = DMPcurv‚ÇÄ .* dimcar .^ 2

    # Then we create a `NamedTuple` to hold the simulation parameters.
    params = (
        degquad = degquad,
        stateInit = stateInit,
        stateBcFarfield = stateBcFarfield,
        DMPrelax = DMPrelax,
    )

    # Define measures for cell and interior face integrations
    dŒ© = Measure(CellDomain(mesh), degquad)
    dŒì = Measure(InteriorFaceDomain(mesh), degquad)

    # Declare boundary conditions and
    # create associated domains and measures
    Œì_wall      = BoundaryFaceDomain(mesh, ("NACA",))
    Œì_farfield  = BoundaryFaceDomain(mesh, ("FARFIELD",))
    dŒì_wall     = Measure(Œì_wall, degquad)
    dŒì_farfield = Measure(Œì_farfield, degquad)

    params = (
        params...,
        Œì_wall = Œì_wall,
        dŒì = dŒì,
        dŒ© = dŒ©,
        dŒì_wall = dŒì_wall,
        dŒì_farfield = dŒì_farfield,
    )

    qLowOrder = nothing

    for deg in 0:degree
        params = (params..., degree = deg)

        fs = FunctionSpace(fspace, deg)
        Q_sca = TrialFESpace(fs, mesh, :discontinuous; size = 1) # DG, scalar
        Q_vec = TrialFESpace(fs, mesh, :discontinuous; size = 2) # DG, vectoriel
        V_sca = TestFESpace(Q_sca)
        V_vec = TestFESpace(Q_vec)
        Q = MultiFESpace(Q_sca, Q_vec, Q_sca)
        V = MultiFESpace(V_sca, V_vec, V_sca)

        q = FEFunction(Q)

        # select an initial configurations:
        if deg == 0
            init!(q, mesh, stateInit)
        else
            println("Start projection")
            projection_l2!(q, qLowOrder, dŒ©)
            println("End projection")
        end

        # create CellData to store limiter values
        limœÅ = Bcube.MeshCellData(ones(ncells(mesh)))
        limAll = Bcube.MeshCellData(ones(ncells(mesh)))
        params = (params..., limœÅ = limœÅ, limAll = limAll)

        # Init vtk handler
        mkpath(outputpath)
        vtk = VtkHandler(
            outputpath * "euler_naca_mdeg" * string(mesh_degree) * "_deg" * string(deg),
        )

        # Init time
        time = 0.0

        # Save initial solution
        append_vtk(vtk, mesh, q, time, params)

        # Build the cache and store everything you want to compute only once (such as the mass matrice inverse...)

        cache = ()
        # Allocate buffer for compute_residual
        b_vol = zeros(Bcube.get_ndofs(Q))
        b_fac = zeros(Bcube.get_ndofs(Q))
        cache = (cache..., b_vol = b_vol, b_fac = b_fac)

        cache = (
            cache...,
            cacheCellMean = Bcube.build_cell_mean_cache(q, dŒ©),
            mass = factorize(Bcube.build_mass_matrix(Q, V, dŒ©)),
            mass_sca = factorize(Bcube.build_mass_matrix(Q_sca, V_sca, dŒ©)),
            mass_vec = factorize(Bcube.build_mass_matrix(Q_vec, V_vec, dŒ©)),
        )

        time, q = steady_solve!(Q, V, q, mesh, params, cache, vtk, deg)
        append_vtk(vtk, mesh, q, time, params)

        deg < degree && (qLowOrder = deepcopy(q))

end

function steady_solve!(Q, V, q, mesh, params, cache, vtk, deg)
    counter = [0]
    q0 = deepcopy(get_dof_values(q))
    ode_params =
        (Q = Q, V = V, params = params, cache = cache, counter = counter, vtk = vtk)

    rhs!(dq, q, p, t) = dq .= compute_residual(q, p.Q, p.V, p.params)

    # compute sparsity pattern and coloring
    println("computing jacobian cache...")
    if withbench
        _rhs!(dq, q) = rhs!(dq, q, ode_params, 0.0)
        @btime $_rhs!(similar($q0), $q0)
        q_bench = FEFunction(Q, q0)
        @btime $apply_limitation!($q_bench, $ode_params)
        @show length(q0)
    end

    #sparsity_pattern = Symbolics.jacobian_sparsity(_rhs!, similar(Q0), Q0)
    #tjac = @elapsed Symbolics.jacobian_sparsity(_rhs!, similar(Q0), Q0)
    #@show tjac
    sparsity_pattern = Bcube.build_jacobian_sparsity_pattern(Q, mesh)
    println("sparsity pattern computed !")
    display(sparsity_pattern)
    colors = matrix_colors(sparsity_pattern)
    println("coloring done!")
    @show maximum(colors)

    ode = ODEFunction(
        rhs!;
        mass_matrix = Bcube.build_mass_matrix(Q, V, params.dŒ©),
        jac_prototype = sparsity_pattern,
        colorvec = colors,
    )

    Tfinal      = Inf
    problem     = ODEProblem(ode, q0, (0.0, Tfinal), ode_params)
    timestepper = ImplicitEuler(; nlsolve = NLNewton(; max_iter = 20))

    cb_cache  = DiscreteCallback(always_true, update_cache!; save_positions = (false, false))
    cb_vtk    = DiscreteCallback(always_true, output_vtk; save_positions = (false, false))
    cb_steady = TerminateSteadyState(1e-6, 1e-6, condition_steadystate)

    error = 1e-1

    sol = solve(
        problem,
        timestepper;
        initializealg = NoInit(),
        adaptive = true,
        abstol = error,
        reltol = error,
        progress = false,
        progress_steps = 1000,
        save_everystep = false,
        save_start = false,
        save_end = false,
        isoutofdomain = isoutofdomain,
        callback = CallbackSet(cb_cache, cb_vtk, cb_steady),
    )

    set_dof_values!(q, sol.u[end])
    return sol.t[end], q
end

function isoutofdomain(dof, p, t)
    any(isnan, dof) && return true

    q = FEFunction(p.Q, dof)
    q_mean = map(get_values, cell_mean(q, p.cache.cacheCellMean))
    p_mean = pressure.(q_mean..., stateInit.Œ≥)

    negative_œÅ = any(x -> x < 0, q_mean[1])
    negative_p = any(x -> x < 0, p_mean)
    isout = negative_œÅ || negative_p
    isout && @show negative_œÅ, negative_p
    return isout
end

function output_vtk(integrator)
    u_modified!(integrator, false)
    mesh = get_mesh(get_domain(integrator.p.params.dŒ©))
    q = FEFunction(integrator.p.Q, integrator.u)
    counter = integrator.p.counter
    counter .+= 1
    if (counter[1] % nout == 0)
        println("output_vtk ", counter[1])
        append_vtk(integrator.p.vtk, mesh, q, integrator.t, integrator.p.params)
    end
    return nothing
end

function condition_steadystate(integrator, abstol, reltol, min_t)
    u_modified!(integrator, false)
    if DiffEqBase.isinplace(integrator.sol.prob)
        testval = first(get_tmp_cache(integrator))
        @. testval = (integrator.u - integrator.uprev) / (integrator.t - integrator.tprev)
    else
        testval = (integrator.u - integrator.uprev) / (integrator.t - integrator.tprev)
    end

    (integrator.p.counter[1] ‚â• maxiters) && (return true)

    if typeof(integrator.u) <: Array
        any(
            abs(d) > abstol && abs(d) > reltol * abs(u) for (d, abstol, reltol, u) in
            zip(testval, Iterators.cycle(abstol), Iterators.cycle(reltol), integrator.u)
        ) && (return false)
    else
        any((abs.(testval) .> abstol) .& (abs.(testval) .> reltol .* abs.(integrator.u))) &&
            (return false)
    end

    if min_t === nothing
        return true
    else
        return integrator.t >= min_t
    end
end

function compute_dimcar(mesh)
    fs = FunctionSpace(:Lagrange, 0)
    V = TestFESpace(fs, mesh; size = 1, isContinuous = false)

    # Define measures for cell and interior face integrations
    dŒ© = Measure(CellDomain(mesh), degquad)
    dŒì = Measure(InteriorFaceDomain(mesh), degquad)
    dŒì_bc = Measure(BoundaryFaceDomain(mesh), degquad)

    f1 = PhysicalFunction(x -> 1.0)
    l(v) = ‚à´(f1 ‚ãÖ v)dŒ©
