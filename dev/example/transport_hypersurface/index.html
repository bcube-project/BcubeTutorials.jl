Transport equation on hypersurfaces

In this file, a linear transport equation is solved on several hypersurfaces. The scalar transport equation is

\[ \partial_t u + c \cdot \nabla_\Gamma u = 0\]

where $c$ is the (tangential) transport velocity (always divergent-free in these examples); and $\nabla_\Gamma$ is the tangential gradient. The vector transport equation is

\[ \partial_t u + \nabla_\Gamma (c \otimes u) = 0\]

Here is an example of the transport of a scalar bump on a torus meshed with P2 triangles:

function plot_solution(i, t, u, mesh, xcenters, ycenters, xnodes, ynodes)
    # Build animation
    uplot = var_on_centers(u, mesh)
    lmax = 1.5

    plt = plot(
        [xnodes..., xnodes[1]],
        [ynodes..., ynodes[1]];
        aspect_ratio = :equal,
        primary = false,
        xlim = (-lmax, lmax),
        ylim = (-lmax, lmax),
    )
    annotate!(0, 0.25, "i  = $i")
    annotate!(0, 0, @sprintf "t = %.2e" t)

    if ndims(uplot) == 1
        scatter!(xcenters, ycenters; marker_z = uplot, label = "u", clims = (-1, 1))
        annotate!(0, -0.25, @sprintf "|u|_max = %.2e" maximum(abs.(uplot)))
    elseif ndims(uplot) == 2
        L = maximum(x -> norm(x), eachrow(uplot))
        scale = 0.75
        uplot .*= scale
        quiver!(
            xcenters,
            ycenters;
            quiver = (uplot[:, 1], uplot[:, 2]),
            label = "u",
            xlabel = "x",
            ylabel = "y",
        )
        annotate!(0, -0.25, @sprintf "|u|_max = %.2e" L)
    end

    return plt
end

"""
    rotMat(őłx, őły, őłz)

Build the 3D rotation matrix from the angles given for each axis.
"""
function rotMat(őłx, őły, őłz)
    Rx = @SMatrix[
        1.0 0.0 0.0
        0.0 cos(őłx) sin(őłx)
        0.0 (-sin(őłx)) cos(őłx)
    ]
    Ry = @SMatrix[
        cos(őły) 0.0 (-sin(őły))
        0.0 1.0 0.0
        sin(őły) 0.0 cos(őły)
    ]
    Rz = @SMatrix[
        cos(őłz) sin(őłz) 0.0
        -sin(őłz) cos(őłz) 0.0
        0.0 0.0 1.0
    ]
    return Rx * Ry * Rz
end

"""
Linear transport of a scalar quantity on a circle

nrot = number of "rotations" to run (in time)
volumic_bilinear = true means the volumic term is assembled once with a bilinear assembly while false means
    a linear assembly (and hence the limiter is applied)
"""
function scalar_circle(;
    degree,
    CFL,
    nőł,
    nrot = 2,
    nout = 100,
    nitemax = Int(1e9),
    volumic_bilinear = false,
    isLimiterActive = true,
)
    function append_vtk(vtk, u::Bcube.AbstractFEFunction, lim_u, u_mean, t)
        # Build animation
        values_centers = var_on_centers(u, mesh)

        # Write
        write_file(
            vtk.basename * ".pvd",
            vtk.mesh,
            Dict(
                "u_centers" => MeshCellData(values_centers),
                "u_vertices" => u,
                "lim_u" => lim_u,
                "u_mean" => u_mean,
                "őł_centers" => vtk.őł_centers,
                "őł_vertices" => vtk.őł,
            ),
            vtk.ite,
            t,
            ;
            collection_append = vtk.ite > 0,
        )

        # Update counter
        vtk.ite += 1
    end

    # Settings
    radius = 1.0
    C = 1.0 ## velocity norm
    tmax = nrot * 2ŌÄ / C

    # Mesh
    qOrder = 2 * degree + 1 ## we shall use Gauss-Lobatto for degree > 0, but in 1D this is ok
    mesh = circle_mesh(nőł; radius = radius, order = 1)
    dő© = Measure(CellDomain(mesh), qOrder)
    őď = InteriorFaceDomain(mesh)
    dőď = Measure(őď, qOrder)
    nőď = get_face_normals(őď)

    # FESpace
    fs = FunctionSpace(:Lagrange, degree)
    U = TrialFESpace(fs, mesh, :discontinuous)
    V = TestFESpace(U)

    # Transport velocity
    _c = PhysicalFunction(x -> C * SA[-x[2], x[1]] / radius)
    P = Bcube.tangential_projector()
    c = (x -> C * normalize(x)) ‚ąė (P * _c) ## useless in theory since velocity is already tangent

    # Find quadrature weight (mesh is composed of a unique "shape" so first element is enough)
    quad = Bcube.get_quadrature(dő©)
    s = Bcube.shape(Bcube.cells(mesh)[1])
    qrule = Bcube.QuadratureRule(s, quad)
    ŌČ_quad = degree > 0 ? Bcube.get_weights(qrule)[1] : 1.0

    # Time step and else
    dl = 2ŌÄ * radius / nőł ## analytic length
    dl = 2 * radius * sin(2ŌÄ / nőł / 2) ## discretized length
    őĒt = CFL * dl * ŌČ_quad / C / (2 * degree + 1)
    t = 0.0
    nite = min(floor(Int, tmax / őĒt), nitemax)
    @show nite
    _nout = min(nite, nout)

    @show dl
    @show őĒt

    # Limitation
    DMPrelax = 0.0 * dl

    # Output
    tail = isLimiterActive ? "lim" : "nolim"
    filename = "scalar-on-circle-d$(degree)-$(tail)"
    vtk = VtkHandler(joinpath(out_dir, filename), dő©, U, c)
    dofOverTime = zeros(nite + 1, 2) ## t, u
    i_dof_out = 1

    # FEFunction and "boundary / source" condition
    u = FEFunction(U)
    if false
        u.dofValues[1] = 1.0
    else
        projection_l2!(u, PhysicalFunction(x -> cos(atan(x[2], x[1]))), mesh)
    end

    # Forms
    m(u, v) = ‚ąę(u ‚čÖ v)dő© ## Mass matrix
    a_ő©(u, v) = ‚ąę(u * (c ‚čÖ ‚ąá‚āõ(v)))dő© ## bilinear volumic convective term

    function upwind(ui, uj, ci, nij)
        cij = ci ‚čÖ nij
        if cij > zero(cij)
            fij = cij * ui
        else
            fij = cij * uj
        end
        fij
    end

    # Mass
    M = assemble_bilinear(m, U, V)
    invM = inv(Matrix(M)) ##WARNING : really expensive !!!
    if volumic_bilinear
        K = assemble_bilinear(a_ő©, U, V)
        invMK = invM * K
    end

    # Anim
    anim = Animation()
    xnodes = [get_coords(node, 1) for node in get_nodes(mesh)]
    ynodes = [get_coords(node, 2) for node in get_nodes(mesh)]
    xcenters = [center[1] for center in Bcube.get_cell_centers(mesh)]
    ycenters = [center[2] for center in Bcube.get_cell_centers(mesh)]

    # Initial solution
    lim_u, _u = linear_scaling_limiter(u, dő©; DMPrelax, mass = M)
    isLimiterActive && (u.dofValues .= _u.dofValues)

    u_mean = cell_mean(u, dő©)
    t = 0.0
    plt = plot_solution(0, t, u, mesh, xcenters, ycenters, xnodes, ynodes)
    append_vtk(vtk, u, lim_u, u_mean, t)
    frame(anim, plt)
    dofOverTime[1, :] .= t, u.dofValues[i_dof_out]

    b = Bcube.allocate_dofs(U)
    for ite in 1:nite
        b .= 0.0

        # Apply limitation
        if isLimiterActive
            lim_u, _u = linear_scaling_limiter(u, dő©; DMPrelax, mass = M)
            set_dof_values!(u, get_dof_values(_u))
        end

        # Define linear forms
        flux = upwind ‚ąė (side‚ĀĽ(u), side‚Āļ(u), side‚ĀĽ(c), side‚ĀĽ(nőď))
        l_őď(v) = ‚ąę(-flux * jump(v))dőď
        l_ő©(v) = ‚ąę(u * (c ‚čÖ ‚ąá‚āõ(v)))dő© ## linear Volumic convective term
        l(v) = l_ő©(v) + l_őď(v)

        if volumic_bilinear
            # Version bilinear volumic term
            assemble_linear!(b, l_őď, V)
            u.dofValues .= (I + őĒt .* invMK) * u.dofValues + őĒt .* invM * b
        else
            # Version linear volumic term
            assemble_linear!(b, l, V)
            u.dofValues .+= őĒt .* invM * b
        end

        t += őĒt

        # Output results
        if ite % (nite √∑ _nout) == 0
            u_mean = cell_mean(u, dő©)
            append_vtk(vtk, u, lim_u, u_mean, t)
            plt = plot_solution(vtk.ite, t, u, mesh, xcenters, ycenters, xnodes, ynodes)
            frame(anim, plt)
        end
        dofOverTime[ite + 1, :] .= t, u.dofValues[i_dof_out]
    end

    # Output final result and anim
    path = joinpath(out_dir, filename * ".csv")
    @info "Writing to $path"
    open(path, "w") do io
        println(io, "t,u")
        writedlm(io, dofOverTime, ",")
    end
    #! format: off
    println("Computation is done, building gif...")
    g = gif(anim, joinpath(out_dir, "$filename.gif"); fps = 4)
    display(g)
    #! format: on

end

"""
Linear transport of a vector quantity on a circle
"""
function vector_circle(; degree, nite, CFL, nőł)
    # Settings
    radius = 1.0
    C = 1.0 ## velocity norm

    # Mesh
    mesh = circle_mesh(nőł; radius = radius, order = 1)
    dő© = Measure(CellDomain(mesh), 2 * degree + 1)
    őď = InteriorFaceDomain(mesh)
    dőď = Measure(őď, 2 * degree + 1)
    nőď = get_face_normals(őď)

    # Operators
    P = Bcube.tangential_projector()
    R = Bcube.CoplanarRotation()

    # Transport velocity : it must be coplanar to each element, so we use the
    # tangential projector operator and force the projected velocity to have
    # the same norm as the "analytical" one
    _c = PhysicalFunction(x -> C * SA[-x[2], x[1]] / radius)
    c = (x -> C * normalize(x)) ‚ąė (P * _c) ## useless in theory since velocity is already tangent

    # FESpace
    fs = FunctionSpace(:Lagrange, degree)
    U = TrialFESpace(fs, mesh, :discontinuous; size = 2)
    V = TestFESpace(U)

    # FEFunction and "boundary / source" condition
    u = FEFunction(U)

    # Initial condition
    _u0 = PhysicalFunction(x -> begin
        őł = atan(x[2], x[1])
        if 0 <= őł <= 2ŌÄ / nőł
            return radius * SA[-x[2], x[1]]
        else
            return SA[0, 0]
        end
    end)
    projection_l2!(u, _u0, mesh)

    # Forms
    m(u, v) = ‚ąę(u ‚čÖ v)dő© ## Mass matrix
    l_ő©(v) = ‚ąę((u ‚äó c) ‚ä° ‚ąá‚āõ(v))dő© ## Volumic convective term

    function upwind(ui, uj, Ri, Rj, vi, vj, ci, nij)
        _uj = Ri * uj

        cij = ci ‚čÖ nij
        if cij > zero(cij)
            fi = cij * ui
        else
            fi = cij * _uj
        end

        return fi ‚čÖ (vi - transpose(Rj) * vj)
    end

    function flux(v)
        upwind ‚ąė
        (side‚ĀĽ(u), side‚Āļ(u), side‚ĀĽ(R), side‚Āļ(R), side‚ĀĽ(v), side‚Āļ(v), side‚ĀĽ(c), side‚ĀĽ(nőď))
    end
    l_őď(v) = ‚ąę(-flux(v))dőď

    l(v) = l_ő©(v) + l_őď(v)

    # Time step
    dl = 2ŌÄ * radius / nőł ## analytic length
    dl = 2 * radius * sin(2ŌÄ / nőł / 2) ## discretized length
    őĒt = CFL * dl / C
    @show dl
    @show őĒt

    # Mass
    M = assemble_bilinear(m, U, V)
    invM = inv(Matrix(M)) ##WARNING : really expensive !!!
    # display(invM)

    # Anim
    anim = Animation()
    xnodes = [get_coords(node, 1) for node in get_nodes(mesh)]
    ynodes = [get_coords(node, 2) for node in get_nodes(mesh)]
    xcenters = [center[1] for center in Bcube.get_cell_centers(mesh)]
    ycenters = [center[2] for center in Bcube.get_cell_centers(mesh)]

    # Initial solution
    plt = plot_solution(0, 0.0, u, mesh, xcenters, ycenters, xnodes, ynodes)
    frame(anim, plt)

    t = 0.0
    b = Bcube.allocate_dofs(U)
    for i in 1:nite
        b .= 0.0
        assemble_linear!(b, l, V)

        u.dofValues .+= őĒt .* invM * b

        t += őĒt

        # Build animation
        plt = plot_solution(i, t, u, mesh, xcenters, ycenters, xnodes, ynodes)
        frame(anim, plt)
    end

    #! format: off
    g = gif(anim, joinpath(out_dir, "vector_on_circle_d$degree.gif"); fps = 2)
    display(g)
    #! format: on

end

"""
Linear transport of a scalar quantity on a cylinder
"""
function scalar_cylinder(;
    degree,
    CFL,
    lz,
    nz,
    nőł,
    tmax,
    Ōē, ## velocity angle with respect to z axis
    C, ## velocity norm
    radius = 1,
    nout = 100,
    nitemax = Int(1e9),
    isLimiterActive = true,
    progressBar = true,
    meshOrder = 1,
    profile = false,
)
    function append_vtk(vtk, u::Bcube.AbstractFEFunction, lim_u, t)
        vars = Dict(
            "u" => u,
            "u_mean" => cell_mean(u, vtk.dő©),
            "lim_u" => lim_u,
            "c" => vtk.c,
            "cellnormal" => vtk.őĹ,
            "u_warp" => u * vtk.őĹ,
        )
        write_file(
            vtk.basename * "_lag.pvd",
            vtk.mesh,
            vtk.U,
            vars,
            vtk.ite,
            t;
            collection_append = vtk.ite > 0,
        )

        # Update counter
        vtk.ite += 1
    end

    # Mesh
    mesh_path = joinpath(out_dir, "mesh.msh")
    BcubeGmsh.gen_cylinder_shell_mesh(
        mesh_path,
        nőł,
        nz;
        lz,
        radius,
        lc = 1e-1,
        recombine = true,
        transfinite = true,
        order = meshOrder,
    )
    mesh = read_mesh(mesh_path)
    rng = Random.MersenneTwister(33)
    őł = rand(rng, 3) .* 2ŌÄ
    println("őłx, őły, őłz = $(rad2deg.(őł))")
    Rmat = rotMat(őł...)
    RmatInv = inv(Rmat)
    transform!(mesh, x -> Rmat * x)

    # Domains
    # quad = Quadrature(QuadratureLobatto(), 2 * degree + 1)
    quad = Quadrature(QuadratureLegendre(), 2 * degree + 1)
    dő© = Measure(CellDomain(mesh), quad)
    őď = InteriorFaceDomain(mesh)
    dőď = Measure(őď, quad)
    nőď = get_face_normals(őď)
    őď_bnd = BoundaryFaceDomain(mesh, ("zmin", "zmax"))
    dőď_bnd = Measure(őď_bnd, quad)
    nőď_bnd = get_face_normals(őď_bnd)

    # FESpace
    fs = FunctionSpace(:Lagrange, degree)
    U = TrialFESpace(fs, mesh, :discontinuous)
    V = TestFESpace(U)

    # Transport velocity
    Cz = C * cos(Ōē)
    Cőł = C * sin(Ōē)
    _c = PhysicalFunction(x -> begin
        _x = RmatInv * x
        Rmat * SA[-Cőł * _x[2] / radius, Cőł * _x[1] / radius, Cz]
    end)
    P = Bcube.tangential_projector()
    c = (x -> C * normalize(x)) ‚ąė (P * _c)

    # Find quadrature weight (mesh is composed of a unique "shape" so first element is enough)
    quad = Bcube.get_quadrature(dő©)
    s = Bcube.shape(Bcube.cells(mesh)[1])
    qrule = Bcube.QuadratureRule(s, quad)
    ŌČ_quad = degree > 0 ? Bcube.get_weights(qrule)[1] : 1.0

    # Time step and else
    dlőł = 2ŌÄ * radius / nőł ## analytic length
    dlőł = 2 * radius * sin(2ŌÄ / nőł / 2) ## discretized length
    dlz = lz / (nz - 1)
    println("Timestep constrained by $(dlőł < dlz ? "őł" : "z") discretization")
    dl = min(dlőł, dlz)
    őĒt = CFL * dl * ŌČ_quad / C / (2 * degree + 1)
    t = 0.0
    nite = min(floor(Int, tmax / őĒt), nitemax)
    _nout = min(nite, nout)

    @show nite
    @show dl
    @show őĒt
    @show get_ndofs(U)

    # Limitation
    DMPrelax = 0.0 * dl

    # Output
    tail = isLimiterActive ? "lim" : "nolim"
    filename = "scalar-on-cylinder-d$(degree)-$(tail)"
    U_export =
        TrialFESpace(FunctionSpace(:Lagrange, max(degree, meshOrder)), mesh, :discontinuous)
    vtk = VtkHandler(joinpath(out_dir, filename), dő©, U_export, c)

    # FEFunction and initial solution (P3 Gaussian bump)
    u = FEFunction(U)
    _őł0 = 0
    x0 = Rmat * SA[radius * cos(_őł0), radius * sin(_őł0), 0.2 * lz] ## bump center (in rotated frame)
    _r = 1 ## bump radius
    _umax = 1 ## bump amplitude
    _a, _b = SA[
        _r^3 _r^2
        3*_r^2 2*_r
    ] \ SA[-_umax; 0]
    f = PhysicalFunction(x -> begin
        dx = norm(x - x0)
        dx < _r ? _a * dx^3 + _b * dx^2 + _umax : 0.0
    end)

    projection_l2!(u, f, mesh)

    # Forms
    m(u, v) = ‚ąę(u ‚čÖ v)dő© ## Mass matrix

    function upwind(ui, uj, ci, nij)
        cij = ci ‚čÖ nij
        if cij > zero(cij)
            fij = cij * ui
        else
            fij = cij * uj
        end
        fij
    end

    # Mass
    M = factorize(assemble_bilinear(m, U, V))

    # Initial solution
    lim_u, _u = linear_scaling_limiter(u, dő©; DMPrelax, mass = M)
    isLimiterActive && (u.dofValues .= _u.dofValues)

    t = 0.0
    append_vtk(vtk, u, lim_u, t)

    b = Bcube.allocate_dofs(U)
    du = similar(b)
    progress = Progress(nitemax; enabled = progressBar)
    for ite in 1:nitemax
        b .= 0.0

        # Apply limitation
        if isLimiterActive
            lim_u, _u = linear_scaling_limiter(u, dő©; DMPrelax, mass = M)
            set_dof_values!(u, get_dof_values(_u))
        end

        # Define linear forms
        flux = upwind ‚ąė (side‚ĀĽ(u), side‚Āļ(u), side‚ĀĽ(c), side‚ĀĽ(nőď))
        l_őď(v) = ‚ąę(-flux * jump(v))dőď
        flux_bnd = upwind ‚ąė (side‚ĀĽ(u), side‚ĀĽ(u), side‚ĀĽ(c), side‚ĀĽ(nőď_bnd))
        l_őď_bnd(v) = ‚ąę(-flux_bnd * side‚ĀĽ(v))dőď_bnd
        l_ő©(v) = ‚ąę(u * (c ‚čÖ ‚ąá‚āõ(v)))dő© ## linear Volumic convective term
        l(v) = l_ő©(v) + l_őď(v) + l_őď_bnd(v)

        # Version linear volumic term
        assemble_linear!(b, l, V)
        du .= M \ b
        @. u.dofValues += őĒt * du

        t += őĒt
        next!(progress)

        # Output results
        if ite % (nitemax √∑ _nout) == 0
            append_vtk(vtk, u, lim_u, t)
        end

        if ite == nitemax && profile
            println("ndofs total = ", Bcube.get_ndofs(U))
            Profile.init(; n = 10^7) ## returns the current settings
            Profile.clear()
            Profile.clear_malloc_data()
            @profile begin
                for i in 1:1000
                    assemble_linear!(b, l, V)
                end
            end
            @btime assemble_linear!($b, $l, $V)
        end
    end

        test_ref(
            "transport_hypersurface_scalar_cylinder_u.jld2",
            get_dof_values(u),
            compare(; rtol = 1e-6),
end
